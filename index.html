import numpy as np
import pandas as pd
import cv2
from sklearn.model_selection import train_test_split

# Load the dataset
data = df

# Preprocess pixel data
def preprocess_pixels(pixel_series, target_size=(64, 64)):
    images = []
    for pixel_str in pixel_series:
        pixel_array = np.array([int(p) for p in pixel_str.split()], dtype=np.uint8)
        pixel_image = pixel_array.reshape(48, 48)
        resized_image = cv2.resize(pixel_image, target_size)
        images.append(resized_image.reshape(target_size[0], target_size[1], 1))
    return np.array(images)

# Prepare data
X = preprocess_pixels(data['pixels'], target_size=(64, 64))
X = np.repeat(X, 3, axis=-1)  # Convert grayscale to RGB
X = X / 255.0  # Normalize pixel values

# Extract target labels
y_age = data['age'].values  # Regression
y_gender = data['gender'].values  # Binary classification: 0 (Female), 1 (Male)
y_ethnicity = data['ethnicity'].values  # Multi-class classification

# Split data into train and test sets
X_train, X_test, y_age_train, y_age_test = train_test_split(X, y_age, test_size=0.2, random_state=42)
_, _, y_gender_train, y_gender_test = train_test_split(X, y_gender, test_size=0.2, random_state=42)
_, _, y_ethnicity_train, y_ethnicity_test = train_test_split(X, y_ethnicity, test_size=0.2, random_state=42)

# Save preprocessed data
np.savez_compressed('preprocessed_dataset.npz',
                    X_train=X_train, X_test=X_test,
                    y_age_train=y_age_train, y_age_test=y_age_test,
                    y_gender_train=y_gender_train, y_gender_test=y_gender_test,
                    y_ethnicity_train=y_ethnicity_train, y_ethnicity_test=y_ethnicity_test)
print("Preprocessed dataset saved successfully.")

# Function to plot training history
def plot_history(history, metric, title):
    plt.plot(history.history[metric], label=f'Training {metric}')
    plt.plot(history.history[f'val_{metric}'], label=f'Validation {metric}')
    plt.title(title)
    plt.xlabel('Epochs')
    plt.ylabel(metric.capitalize())
    plt.legend()
    plt.show()
